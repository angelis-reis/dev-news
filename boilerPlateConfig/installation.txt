Install this libs and make this configurations

npm install eslint --save-dev

./node_modules/.bin/eslint --init
// configure it using AIRBNB

npm install eslint-plugin-only-warn --save-dev

npm install --save-dev --save-exact prettier

npm install --save-dev eslint-config-prettier


Copy the config files to the root folder (overwriting the generated .eslintrc.json )

// In old projects, to format all files run: npx prettier --write .


if it is not already, add it in settings.json (open in the View - Command Pallete)

"editor.codeActionsOnSave": {
      "source.fixAll.eslint": true
    },
